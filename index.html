WhatsApp Bot
Message to add expenses automatically
Bot Status
⏳ Not configured yet
How to use
Send a message like this:
Ramesh 5000 cement guravaiah
Subba 1200 food aunty mess
R 15000 labour sai mestri
S 800 fuel diesel pullaiah
Format: Partner Amount Category Person
Setup Guide
1
Create a free Twilio account
Go to twilio.com → sign up free. You get a trial number that works with WhatsApp immediately.
2
Enable WhatsApp Sandbox
Twilio Console → Messaging → Try it out → Send a WhatsApp message.
3
Deploy the bot (free)
Use the code below on Railway.app or Render.com — both free forever for small apps.
4
Set webhook URL in Twilio
Paste your Railway/Render URL into Twilio's WhatsApp sandbox webhook. Done!
// npm install express twilio @supabase/supabase-js const express = require('express'); const twilio = require('twilio'); const { createClient } = require('@supabase/supabase-js'); const app = express(); app.use(express.urlencoded({ extended: false })); // ⚠️ Use your ANON key here (NOT the service_role key). // The service_role key bypasses RLS and must never be in client code. // Find your anon key: Supabase dashboard → Project Settings → API const sb = createClient('YOUR_SUPABASE_URL', 'YOUR_SUPABASE_ANON_KEY'); const CATS = { cement:'Cement',sand:'Aggregates & Materials',steel:'Steel & Iron',labour:'Labour & Wages',food:'Food & Meals',fuel:'Fuel',water:'Water Supply',jcb:'Equipment / Machinery',advance:'Advance / Hand Loan' }; function today() { const d=new Date(); return `${String(d.getDate()).padStart(2,'0')}/${String(d.getMonth()+1).padStart(2,'0')}/${d.getFullYear()}`; } app.post('/wa-bot', async (req, res) => { const msg=(req.body.Body||'').trim().toLowerCase(); const twiml=new twilio.twiml.MessagingResponse(); const parts=msg.split(/\s+/); const partner=parts[0]; const amount=parseFloat(parts[1]); const catKey=parts[2]||''; const person=parts.slice(3).join(' ')||'Unknown'; const p=['s','subba'].includes(partner)?'S':['r','ramesh'].includes(partner)?'R':null; if(!p||isNaN(amount)||amount<=0){ twiml.message('❌ Format: Partner Amount Category Person'); return res.type('text/xml').send(twiml.toString()); } const {error}=await sb.from('expenses').insert([{date:today(),partner:p,person,description:catKey,amount,category:CATS[catKey]||'Other'}]); twiml.message(error?'❌ '+error.message:`✅ Saved! ${p==='S'?'🟦':'🟧'} ₹${amount.toLocaleString('en-IN')} · ${CATS[catKey]||'Other'} · ${today()}`); res.type('text/xml').send(twiml.toString()); }); app.listen(process.env.PORT||3000,()=>console.log('Bot running!'));
⚠️ Use your Supabase Service Role Key in the bot server — never the anon key, and never in the frontend.
